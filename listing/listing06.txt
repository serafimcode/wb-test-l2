Что выведет программа? Объяснить вывод программы. Рассказать про внутреннее устройство слайсов и что происходит при передаче их в качестве аргументов функции.

package main

import (
  "fmt"
)

func main() {
  var s = []string{"1", "2", "3"}
  modifySlice(s)
  fmt.Println(s)
}

func modifySlice(i []string) {
// В локальной переменной i содержится зн-е слайса, который содержит ссылку на массив {1,2,3}
// Изменяем массив, на который ссылается слайс, теперь в содержится слайс с ссылкой на массив {3,2,3}
  i[0] = "3"
   // Дальнейшие изменения после этого шага нам не интересны, т.к. append возвращает новый слайс,
   // теперь в i лежит новый слайс с ссылкой на новый массив, соответственно изначальный массив больше не меняется
  i = append(i, "4")
  i[1] = "5"
  i = append(i, "6")
}
